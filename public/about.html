import express from 'express'; import path from 'path'; import { fileURLToPath } from 'url'; const __filename = fileURLToPath(import.meta.url); const __dirname = path.dirname(__filename); const app = express(); const PORT = process.env.PORT || 10000; // Si NO quieres usar /public, comenta la línea estática: // app.use(express.static(path.join(__dirname, 'public'))); app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'index.html')); // raíz del repo }); app.listen(PORT, () => { console.log(`Server listening on ${PORT}`); });
